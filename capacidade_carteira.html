WITH ativas AS (
  SELECT
    d.id_ax,
    d.proprietario_b1_email,
    -- fullname pode vir nulo; usa name como fallback
    COALESCE(
      NULLIF(TRIM(d.proprietario_b1_fullname), ''),
      NULLIF(TRIM(d.proprietario_b1_name), '')
    ) AS nome_b1_raw,
    CASE
      WHEN LOWER(d.segmentacao_atendimento) IN ('high touch', 'high_touch') AND d.baldinho = 'Renovação'
        THEN 'High touch - Renovação'
      WHEN LOWER(d.segmentacao_atendimento) IN ('high touch', 'high_touch')
        THEN 'High touch - Novas'
      WHEN LOWER(d.segmentacao_atendimento) IN ('medium touch', 'medium_touch')
        THEN 'Medium touch'
      WHEN LOWER(d.segmentacao_atendimento) IN ('premium touch', 'premium_touch')
        THEN 'Premium touch'
      WHEN LOWER(d.segmentacao_atendimento) IN ('premium touch 40+', 'premium_touch_40+')
        THEN 'Premium touch 40+'
      ELSE 'Outros'
    END AS seg_atual,
    CASE
      WHEN LOWER(d.segmentacao_atendimento) IN ('high touch', 'high_touch') AND d.baldinho = 'Renovação'
        THEN 'High touch - Renovação'
      WHEN LOWER(d.segmentacao_atendimento) IN ('high touch', 'high_touch')
        THEN 'High touch - Novas'
      WHEN LOWER(d.segmentacao_atendimento) IN ('medium touch', 'medium_touch')
        THEN 'Medium touch'
      WHEN LOWER(d.segmentacao_atendimento) IN ('premium touch', 'premium_touch')
        THEN 'Premium touch'
      WHEN LOWER(d.segmentacao_atendimento) IN ('premium touch 40+', 'premium_touch_40+')
        THEN 'Premium touch 40+'
      ELSE 'Outros'
    END AS seg_corrigida
  FROM `lakehouse-378716.crm_ax_raw.ax_deals_temp_` d
  WHERE d.pais = 'Brasil'
    AND d.ano_jornada = '2026'
    AND (d.etapa IN ('Ciclo de Vendas', 'Ciclo de Eventos', 'Onboarding & Boas vindas', 'Em Aplicação', 'Retração')
       OR (d.etapa = 'Enviou Lista de Alunos' AND d.tipo_de_entrada = 'self service'))
    AND d.segmentacao_atendimento NOT IN ('Low_touch')
    AND (
      (d.segmentacao_atendimento IN ('High touch', 'Medium touch') AND d.data_da_reuniao_bv_concluida IS NOT NULL)
      OR d.segmentacao_atendimento IN ('Premium touch', 'Premium touch 40+')
    )
    AND (data_do_evento_de_autografos IS NULL OR CAST(NULLIF(TRIM(data_do_evento_de_autografos), '') AS DATE) > CURRENT_DATE())
),
-- Conta divergências para exibir na nota da visão corrigida
div AS (
  SELECT COUNTIF(seg_atual != seg_corrigida) AS n_div FROM ativas
)
SELECT
  a.proprietario_b1_email,
  a.nome_b1_raw,
  a.seg_atual,
  a.seg_corrigida,
  COUNT(*) AS n_escolas,
  (SELECT n_div FROM div) AS total_divergencias
FROM ativas a
GROUP BY 1, 2, 3, 4
ORDER BY seg_atual, n_escolas DESC
Metodologia: Capacidade máxima por consultor — 80 escolas para Analistas e 60 escolas para Estagiários. Considera apenas consultores ativos (excluindo Supervisores e Coordenação).
Escolas ativas: Brasil · 2026 · etapas Enviou Lista de Alunos (self service), Onboarding & Boas vindas, Em Aplicação, Ciclo de Vendas e Ciclo de Eventos · excluindo Low touch. Escolas com evento de autógrafos já realizado são excluídas.
Classificação do consultor pelo time (time_de_sucesso.md); HT separado em Novas e Renovação pelo campo baldinho: HT Renovação = baldinho = ‘Renovação’ · HT Novas = demais valores ou nulo.

Saldo positivo = capacidade disponível · Saldo negativo = necessidade de contratação (estimativa: 1 Estagiário = 60 escolas).
